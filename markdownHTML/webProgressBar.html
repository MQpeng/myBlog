jquery监听滚动事件


  $(document).ready(function() {
    // 监听内容的滚动事件
    $("#content").scroll(function() {
      var progress =
        $(this).context.scrollTop / ($(this).context.scrollHeight - $(this).context.offsetHeight);
      if (progress >= 1) {
        progress = 1;
      }
      var width = document.body.clientWidth * progress;
      console.log(width);
      // 设置 加载条 的宽度
      $("#loading-progress").width(width);
    });
  });
